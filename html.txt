function pegar(cat,item){
  let nome = prompt("Digite seu nome:");
  if(!nome) return;
  db.ref("itens/"+cat+"/"+item).set(nome);
}

function render(){
  db.ref("itens").on("value", snap=>{
    let dados = snap.val() || {};
    let html = "";

    for(let cat in lista){
      html += `<div class="categoria"><h2>${cat}</h2></div>`;

      lista[cat].forEach(item=>{
        let pessoa = dados?.[cat]?.[item];

        html += `
          <div class="item">
            <b>${item}</b><br>
            ${
              pessoa
              ? `<span style="color:green">${pessoa} vai levar ✅</span>`
              : `<button onclick="pegar('${cat}','${item}')">Vou levar</button>`
            }
          </div>
        `;
      });
    }

    document.getElementById("lista").innerHTML = html;
  });
}

function admin(){
  let senha = prompt("Senha:");
  if(senha !== "1234") return alert("Senha incorreta");

  let item = prompt("Nome do item para liberar novamente:");
  let cat = prompt("Categoria:");

  if(item && cat){
    db.ref("itens/"+cat+"/"+item).remove();
  }
}
